021public final class Version {
022    // the below tokens are replaced during the CI build process. Only the first instance of the token is replaced,
023    // so if needed, additional checking or token comparison can occur further down in the code.
024
025    private static final String buildnumber = "138"; // updated during the CI build process
026    private static final String version = "1.1.138"; // updated during the CI build process
027    private static final String commit = "8ad01018519d9df98a31f227b4c38eada62b80c5"; // updated during the CI build process
028
029    private static String appBuildNumber = "unknown";
030    private static String appVersionNumber = "unknown";
031    private static String appPackageName = "unknown";
032
033    public static final String Version;
034    public static final String BuildVersion;
035    public static final String Commit;
036
037    private Version() {}
038
039    static {
040        Version = version;
041        BuildVersion = version + "." + buildnumber;
042        Commit = commit;
043
044        addVersionAddendum();
045        new Version();
046    }
047
048    private static void addVersionAddendum() {
049        PredixSDKLogger.trace(Version.getClass().getName(), "Initialized AndroidSDK: Version: " + BuildVersion);
050        VersionService.addVersionServiceAddendumProvider(new VersionServiceAddendumProvider() {
051            @Override
052            public Map<String, String> getVersionServiceAddendum() {
053                String buildVersion = com.ge.predix.mobile.sdk.android.Version.appVersionNumber.concat(".").concat(com.ge.predix.mobile.sdk.android.Version.appBuildNumber);
054                Map<String, String> returnMap = new HashMap<String, String>();
055                returnMap.put(PredixConstants.VersionInfoKeys.ApplicationBuildVersion, buildVersion);
056                returnMap.put(PredixConstants.VersionInfoKeys.ApplicationVersion, com.ge.predix.mobile.sdk.android.Version.appVersionNumber);
057                returnMap.put(PredixConstants.VersionInfoKeys.ApplicationBundleId, com.ge.predix.mobile.sdk.android.Version.appPackageName);
058                returnMap.put(PredixConstants.VersionInfoKeys.AndroidSDKVersion, com.ge.predix.mobile.sdk.android.Version.Version);
059                returnMap.put(PredixConstants.VersionInfoKeys.AndroidSDKBuildVersion, com.ge.predix.mobile.sdk.android.Version.BuildVersion);
060                returnMap.put(PredixConstants.VersionInfoKeys.DeviceModel, android.os.Build.MODEL);
061                return returnMap;
062            }
063        });
064    }
065
066    public static void setupVersionAddendum(Context context)
067    {
068        if (context != null) {
069            PackageManager packageManager = context.getPackageManager();
070            if(packageManager != null) {
071                appPackageName = context.getPackageName();
072
073                try {
074                    PackageInfo packageInfo = packageManager.getPackageInfo(appPackageName, 0);
075                    if (packageInfo != null) {
076                        appVersionNumber = packageInfo.versionName;
077                        appBuildNumber = String.valueOf(packageInfo.versionCode);
078                    }
079                } catch (PackageManager.NameNotFoundException e) {
080                    PredixSDKLogger.error("No package information while gathering version.");
081                }
082            }
083        }
084    }
085}
