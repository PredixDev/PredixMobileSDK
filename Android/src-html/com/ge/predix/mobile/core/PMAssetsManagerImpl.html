001package com.ge.predix.mobile.core;
002
003import android.content.Context;
004import android.content.res.AssetManager;
005
006import com.ge.predix.mobile.logging.PredixSDKLogger;
007import com.ge.predix.mobile.platform.PMAssetsManager;
008import com.ge.predix.mobile.util.URLUtil;
009
010import java.io.File;
011import java.io.FileOutputStream;
012import java.io.IOException;
013import java.io.InputStream;
014import java.io.OutputStream;
015import java.util.Arrays;
016import java.util.List;
017
018import javax.inject.Inject;
019

022 */
023public class PMAssetsManagerImpl implements PMAssetsManager {
024
025    private final Context context;
026
027    @Inject
028    public PMAssetsManagerImpl(Context androidContext) {
029        this.context = androidContext;
030    }
031
032    @Override
033    public boolean isFilePresent(String fileName) {
034        boolean toReturn = false;
035        try {
036            List<String> resources = Arrays.asList(this.context.getResources().getAssets().list(""));
037            toReturn = resources.contains(fileName);
038            PredixSDKLogger.debug(this,fileName+": present: "+toReturn);
039        } catch (IOException e) {
040            PredixSDKLogger.error(this,fileName+": Error while finding file!", e);
041        }
042        return toReturn;
043    }
044
045    @Override
046    public void copyAssets(String assetName, String toPath, AssetsCallback callback) {
047        String path = URLUtil.addSeperatorAtTheEnd(toPath);
048        try {
049            copyAssets(assetName, path);
050            callback.onComplete(true, null);
051        } catch (IOException e) {
052            PredixSDKLogger.error(this,"Error while copying the assets!",e);
053            callback.onComplete(false, e);
054        }
055
056    }
057
058    private void copyAssets(String path, String basePath) throws IOException {
059        AssetManager assetManager = this.context.getResources().getAssets();
060        String assets[];
061            assets = assetManager.list(path);
062            if (assets.length == 0) {
063                copyFile(path,basePath);
064            } else {
065                String fullPath =  basePath + path;
066                File dir = new File(fullPath);
067                if (!dir.exists()) {
068                    if (!dir.mkdirs()) {
069                        IOException ioException = new IOException("Unable to create dir: " + path);
070                        PredixSDKLogger.error(this, "Error while creating directory.",ioException);
071                        throw ioException;
072                    }
073                }
074                for (int i = 0; i < assets.length; ++i) {
075                    String newPath;
076                    if (path.equals("")) {
077                        newPath = "";
078                    }
079                    else {
080                        newPath = path + "/";
081                    }
082                    copyAssets( newPath + assets[i], basePath);
083                }
084            }
085
086    }
087
088    private void copyFile(String filename, String basePath) throws IOException {
089        AssetManager assetManager = this.context.getResources().getAssets();
090
091        InputStream sourceFileIS = null;
092        OutputStream destinationFileOS = null;
093        String newFileName = null;
094        try {
095            PredixSDKLogger.trace(this, "copying file: "+filename);
096            sourceFileIS = assetManager.open(filename);
097            newFileName = basePath + filename;
098            destinationFileOS = new FileOutputStream(newFileName);
099
100            byte[] buffer = new byte[8 * 1024]; //BufferedInputStream --> DEFAULT_BUFFER_SIZE = 8192;
101            int read;
102            while ((read = sourceFileIS.read(buffer)) != -1) {
103                destinationFileOS.write(buffer, 0, read);
104            }
105        } catch (IOException e) {
106            PredixSDKLogger.error(this, "Exception in while copying a file! ", e);
107            throw e;
108        }
109        finally {
110            if (sourceFileIS != null) {
111                try {
112                    sourceFileIS.close();
113                } catch (IOException e) {
114                    PredixSDKLogger.error(this,"Error in closing input stream.",e);
115                }
116            }
117            if (destinationFileOS != null) {
118                try {
119                    destinationFileOS.flush();
120                    destinationFileOS.close();
121                } catch (IOException e) {
122                    PredixSDKLogger.error(this,"Error in closing output stream.",e);
123                }
124            }
125        }
126
127    }
128
129    @Override
130    public void removeAssets(String path, AssetsCallback assetsCallback) {
131        String delPath = URLUtil.addSeperatorAtTheEnd(path);
132        File asset = new File(delPath);
133        if (!asset.exists()) {
134            IOException ioException = new IOException("Asset not found!");
135            PredixSDKLogger.debug(this,"Asset not found on given path.", ioException);
136            assetsCallback.onComplete(false, ioException);
137            return;
138        }
139        if (deleteAssets(asset)) {
140            PredixSDKLogger.debug(this,"Removed asset successfully.");
141            assetsCallback.onComplete(true, null);
142        } else {
143            IOException ioException = new IOException("Unable to delete assets!");
144            PredixSDKLogger.error(this,"Unable to delete assets.", ioException);
145            assetsCallback.onComplete(false, ioException);
146        }
147    }
148
149    private boolean deleteAssets(File dir) {
150        File[] assets = dir.listFiles();
151        boolean toReturn = true;
152        if (assets != null) {
153            for (File asset : assets) {
154                if (asset.isDirectory()) {
155                    toReturn &= deleteAssets(asset);
156                }
157                if (!asset.delete()) {
158                    PredixSDKLogger.error(this,"Unable to delete " + asset);
159                    toReturn = false;
160                }
161            }
162        }
163        return toReturn;
164    }
